Schema SQL

Copie e cole no SQL Editor do Supabase para criar as tabelas.

-- LANORT — Schema completo -- Cole no SQL Editor do Supabase CREATE TABLE IF NOT EXISTS produtos ( id BIGSERIAL PRIMARY KEY, codigo TEXT UNIQUE NOT NULL, descricao TEXT NOT NULL, marca TEXT, ncm TEXT, cest TEXT, estoque INTEGER DEFAULT 0, est_transito INTEGER DEFAULT 0, codigo_barra TEXT, preco NUMERIC(10,4) DEFAULT 0, imagem_url TEXT, ativo BOOLEAN DEFAULT TRUE, criado_em TIMESTAMPTZ DEFAULT NOW(), atualizado_em TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS usuarios ( id BIGSERIAL PRIMARY KEY, codigo TEXT UNIQUE NOT NULL, nome TEXT NOT NULL, email TEXT, acesso BOOLEAN DEFAULT TRUE, tipo TEXT DEFAULT 'parceiro', criado_em TIMESTAMPTZ DEFAULT NOW(), atualizado_em TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS prazos ( id BIGSERIAL PRIMARY KEY, tipo TEXT UNIQUE NOT NULL, descricao TEXT, ativo BOOLEAN DEFAULT TRUE, criado_em TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS pedidos ( id BIGSERIAL PRIMARY KEY, numero_pedido TEXT NOT NULL, numero_item TEXT, usuario_codigo TEXT, usuario_nome TEXT, email TEXT, prazo TEXT, observacoes TEXT, codigo_produto TEXT, descricao_prod TEXT, quantidade NUMERIC(10,2), valor_unitario NUMERIC(10,4), valor_total NUMERIC(10,4), status TEXT DEFAULT 'PENDENTE', criado_em TIMESTAMPTZ DEFAULT NOW() ); CREATE INDEX IF NOT EXISTS idx_prod_cod ON produtos(codigo); CREATE INDEX IF NOT EXISTS idx_prod_marca ON produtos(marca); CREATE INDEX IF NOT EXISTS idx_ped_num ON pedidos(numero_pedido); CREATE OR REPLACE FUNCTION upd_ts() RETURNS TRIGGER AS $$ BEGIN NEW.atualizado_em = NOW(); RETURN NEW; END; $$ LANGUAGE plpgsql; CREATE OR REPLACE TRIGGER trg_prod_ts BEFORE UPDATE ON produtos FOR EACH ROW EXECUTE FUNCTION upd_ts(); CREATE OR REPLACE TRIGGER trg_user_ts BEFORE UPDATE ON usuarios FOR EACH ROW EXECUTE FUNCTION upd_ts(); -- RLS ALTER TABLE produtos ENABLE ROW LEVEL SECURITY; ALTER TABLE usuarios ENABLE ROW LEVEL SECURITY; ALTER TABLE prazos ENABLE ROW LEVEL SECURITY; ALTER TABLE pedidos ENABLE ROW LEVEL SECURITY; CREATE POLICY "pub_prod" ON produtos FOR SELECT TO anon USING(ativo=TRUE); CREATE POLICY "pub_user" ON usuarios FOR SELECT TO anon USING(acesso=TRUE); CREATE POLICY "pub_prazo" ON prazos FOR SELECT TO anon USING(ativo=TRUE); CREATE POLICY "ins_ped" ON pedidos FOR INSERT TO anon WITH CHECK(TRUE); CREATE POLICY "adm_prod" ON produtos FOR ALL TO service_role USING(TRUE); CREATE POLICY "adm_user" ON usuarios FOR ALL TO service_role USING(TRUE); CREATE POLICY "adm_prazo" ON prazos FOR ALL TO service_role USING(TRUE);
